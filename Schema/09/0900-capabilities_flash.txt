CREATE TABLE capabilities_flash (
  id            SERIAL8 PRIMARY KEY,
  session_id    INT8 NOT NULL REFERENCES sessions,
  language_id           INT8 NOT NULL REFERENCES languages,
  avHardwareDisable     BOOLEAN NOT NULL,
  hasAccessibility      BOOLEAN NOT NULL,
  hasAudio              BOOLEAN NOT NULL,
  hasAudioEncoder       BOOLEAN NOT NULL,
  hasEmbeddedVideo      BOOLEAN NOT NULL,
  hasIME                BOOLEAN NOT NULL,
  hasMP3                BOOLEAN NOT NULL,
  hasPrinting           BOOLEAN NOT NULL,
  hasScreenBroadcast    BOOLEAN NOT NULL,
  hasScreenPlayback     BOOLEAN NOT NULL,
  hasStreamingAudio     BOOLEAN NOT NULL,
  hasStreamingVideo     BOOLEAN NOT NULL,
  hasTLS                BOOLEAN NOT NULL,
  hasVideoEncoder       BOOLEAN NOT NULL,
  isDebugger            BOOLEAN NOT NULL,
  localFileReadDisable  BOOLEAN NOT NULL,
  manufacturer          VARCHAR(63) NOT NULL,
  os                    VARCHAR(63) NOT NULL,
  pixelAspectRatio      FLOAT NOT NULL, -- ?
  playerType            VARCHAR(63) NOT NULL,
  screenColor           VARCHAR(63) NOT NULL,
  screenDPI             INT4 NOT NULL,
  screenResolutionXY    VARCHAR(63) NOT NULL,
  version               VARCHAR(63) NOT NULL
);
